.model small
.stack 100h
.code
main proc
    
    print 'First Number:'
    mov ah,1  
    int 21h
    mov bl,al      ; bl=9
    

    Printn
    print 'Second number:'
    mov ah,1    
    int 21h        ; al=5
     
    
    add al,bl
    mov ah,0 ;ah stored 0 for clear register
    aaa      ;adjust after addition   
    
    add al,48
    add ah,48
    
    mov bx,ax ;storing value of ax to bx
    
    printn
    print 'Result is:'
    
    mov ah,2
    mov dl,bh
    int 21h
    
    mov ah,2
    mov dl,bl
    int 21h
    
    exit:
    mov ah,4ch
    int 21h
    main endp
end main
